function( compare )
var set result to null
trace( "Testing if/else branching..." )
if compare == 0
	set result to 0
elseif compare == 1
	set result to 1
else
	if compare == 2
		set result to 2
	else
		set result to 3
	end
end
trace( "Compare " + string->compare + ", Result " + string->result )
if ( compare != result ) && not ( result > 2 )
	trace( "--> Failed" )
	return 0
end
trace( "--> Passed" )
return 1
