function( compare )
trace( "Testing if/else branching..." )
if compare == 0
	var set result to 0
elseif compare == 1
	var set result to 1
else
	if compare == 2
		var set result to 2
	else
		var set result to 3
	end
end
trace( "Compare " + string->compare + ", Result " + string->result )
if ( compare != result ) && not ( compare > 2 )
	trace( "--> Failed" )
	return
end
trace( "--> Passed" )
